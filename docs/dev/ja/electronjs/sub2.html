  3. Execute
    1. プロジェクトrootのpackage.json修正
      package.json
      {
    "name": "electronjs_react_study",
    "version": "0.1.0",
    "private": true,
    "main": "public/Main.js",
    "homepage": "./",
    .
    .
    .
    "scripts": {
        "start": "react-scripts start",
        "build": "react-scripts build",
        "test": "react-scripts test",
        "eject": "react-scripts eject",
        "react-start": "yarn start",
        "electron-start": "set ELECTRON_START_URL=http://localhost:3000 && electron .",
        "electron-pack": "yarn build && electron-builder build -c.extraMetadata.main=build/main.js"
    },
    .
    .
    .
}
    2. publicにmain.js作成
      main.js
      const {app, BrowserWindow} = require('electron');
const path = require('path');
const url = require('url');
function createWindow() {
    const win = new BrowserWindow({
        width:1920,
        height:1080
    });
    const startUrl = process.env.ELECTRON_START_URL || url.format({
        pathname: path.join(__dirname, '/../build/index.html'),
        protocol: 'file:',
        slashes: true
    });
    win.loadURL(startUrl);
}

app.on('ready', createWindow);
    3. プロジェクトrootディレクトリに.envファイルを作成
      .env
      BROWSER=none

    4. yarn react-start でReactをバックグラウンドで起動

    5. yarn electron-start でElectronで起動
